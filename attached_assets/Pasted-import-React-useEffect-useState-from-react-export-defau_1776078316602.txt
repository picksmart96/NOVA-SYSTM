import React, { useEffect, useState } from "react";

export default function NovaDemo() {
  const [started, setStarted] = useState(false);
  const [input, setInput] = useState("");
  const [messages, setMessages] = useState([]);

  // 🔊 AUTO NOVA GREETING
  useEffect(() => {
    const speak = (text) => {
      const msg = new SpeechSynthesisUtterance(text);
      window.speechSynthesis.speak(msg);
    };

    setTimeout(() => {
      speak("Hi, I'm NOVA. Ask me anything about warehouse work or PickSmart Academy.");
    }, 1000);
  }, []);

  const sendMessage = async () => {
    const userMsg = input;
    setMessages([...messages, { role: "user", text: userMsg }]);
    setInput("");

    const res = await fetch("/api/nova-help", {
      method: "POST",
      headers: {
        "Content-Type": "application/json",
      },
      body: JSON.stringify({ message: userMsg, isPublic: true }),
    });

    const data = await res.json();

    setMessages((prev) => [...prev, { role: "assistant", text: data.reply }]);

    const msg = new SpeechSynthesisUtterance(data.reply);
    window.speechSynthesis.speak(msg);
  };

  return (
    <div className="min-h-screen bg-slate-950 text-white flex flex-col items-center justify-center p-6">
      <h1 className="text-4xl font-bold mb-6 text-yellow-400">
        PickSmart NOVA
      </h1>

      <p className="mb-6 text-center max-w-xl">
        Click below and ask NOVA anything about warehouse training, picking,
        safety, or performance.
      </p>

      <div className="w-full max-w-xl space-y-4">
        {messages.map((m, i) => (
          <div key={i} className={m.role === "user" ? "text-right" : "text-left"}>
            <p className={m.role === "user" ? "text-yellow-300" : "text-white"}>
              {m.text}
            </p>
          </div>
        ))}

        <div className="flex gap-2">
          <input
            value={input}
            onChange={(e) => setInput(e.target.value)}
            className="flex-1 p-3 rounded bg-slate-800"
            placeholder="Ask NOVA..."
          />
          <button
            onClick={sendMessage}
            className="bg-yellow-400 text-black px-4 rounded"
          >
            Ask
          </button>
        </div>
      </div>

      {/* 💰 CONVERSION BUTTON */}
      <button className="mt-10 bg-green-500 px-6 py-3 rounded text-black font-bold">
        Request Full Access
      </button>
    </div>
  );
}
