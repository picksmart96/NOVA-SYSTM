2) Create src/pages/training/BeginnerBasicsLessonPage.jsx
import { useMemo, useState } from "react";
import { beginnerBasicsLessonPage as lesson } from "../../data/beginnerBasicsLessonPage";

export default function BeginnerBasicsLessonPage() {
  const [currentQuestion, setCurrentQuestion] = useState(0);
  const [answers, setAnswers] = useState({});
  const [submitted, setSubmitted] = useState(false);

  const score = useMemo(() => {
    const correctCount = lesson.quiz.questions.filter(
      (q) => answers[q.id] === q.correctIndex
    ).length;

    return Math.round((correctCount / lesson.quiz.questions.length) * 100);
  }, [answers]);

  const passed = score >= lesson.quiz.passingScore;

  const handleAnswer = (questionId, optionIndex) => {
    if (submitted) return;
    setAnswers((prev) => ({ ...prev, [questionId]: optionIndex }));
  };

  const handleSubmitQuiz = () => {
    setSubmitted(true);
  };

  const question = lesson.quiz.questions[currentQuestion];

  return (
    <div className="min-h-screen bg-slate-950 p-6 text-white">
      <div className="mx-auto max-w-5xl space-y-8">
        <div className="rounded-3xl border border-slate-800 bg-slate-900 p-8">
          <div className="flex flex-wrap items-center gap-3">
            <span className="rounded-full bg-yellow-500/20 px-3 py-1 text-xs font-bold text-yellow-300">
              {lesson.level}
            </span>
            <span className="rounded-full bg-green-500/20 px-3 py-1 text-xs font-bold text-green-300">
              {lesson.badge}
            </span>
            <span className="rounded-full bg-slate-800 px-3 py-1 text-xs font-bold text-slate-300">
              {lesson.duration}
            </span>
          </div>

          <h1 className="mt-4 text-4xl font-black">{lesson.hero.heading}</h1>
          <p className="mt-4 text-lg text-slate-300">{lesson.hero.subheading}</p>

          <div className="mt-6 rounded-2xl border border-slate-800 bg-slate-950 p-5">
            <p className="text-sm font-bold uppercase tracking-wide text-yellow-300">
              NOVA Intro Voice
            </p>
            <p className="mt-3 text-slate-200">{lesson.introVoice}</p>
          </div>
        </div>

        {lesson.sections.map((section) => (
          <div
            key={section.id}
            className="rounded-3xl border border-slate-800 bg-slate-900 p-8"
          >
            <h2 className="text-2xl font-bold">{section.title}</h2>

            <div className="mt-5 space-y-3 text-slate-300">
              {section.body.map((line, index) => (
                <p key={index}>{line}</p>
              ))}
            </div>

            {section.bullets ? (
              <ul className="mt-5 list-disc space-y-2 pl-6 text-slate-300">
                {section.bullets.map((item, index) => (
                  <li key={index}>{item}</li>
                ))}
              </ul>
            ) : null}

            <div className="mt-6 rounded-2xl border border-slate-800 bg-slate-950 p-5">
              <p className="text-sm font-bold uppercase tracking-wide text-yellow-300">
                NOVA Voice
              </p>
              <p className="mt-3 text-slate-200">{section.novaVoice}</p>
            </div>
          </div>
        ))}

        <div className="rounded-3xl border border-slate-800 bg-slate-900 p-8">
          <h2 className="text-3xl font-black">Lesson Test</h2>
          <p className="mt-3 text-slate-300">
            Pass {lesson.quiz.passingScore}% or higher to complete this lesson.
          </p>

          <div className="mt-8 rounded-2xl border border-slate-800 bg-slate-950 p-6">
            <p className="text-sm font-bold uppercase tracking-wide text-yellow-300">
              Question {currentQuestion + 1} of {lesson.quiz.questions.length}
            </p>

            <h3 className="mt-4 text-xl font-bold">{question.question}</h3>

            <div className="mt-6 space-y-3">
              {question.options.map((option, optionIndex) => {
                const selected = answers[question.id] === optionIndex;
                const correct = submitted && question.correctIndex === optionIndex;
                const wrongSelected =
                  submitted && selected && question.correctIndex !== optionIndex;

                return (
                  <button
                    key={optionIndex}
                    onClick={() => handleAnswer(question.id, optionIndex)}
                    className={`w-full rounded-2xl border px-4 py-4 text-left transition ${
                      correct
                        ? "border-green-500 bg-green-500/10 text-green-200"
                        : wrongSelected
                        ? "border-red-500 bg-red-500/10 text-red-200"
                        : selected
                        ? "border-yellow-400 bg-yellow-500/10 text-yellow-200"
                        : "border-slate-700 bg-slate-900 text-slate-200 hover:border-yellow-400"
                    }`}
                  >
                    {option}
                  </button>
                );
              })}
            </div>

            <div className="mt-6 flex flex-wrap gap-3">
              <button
                onClick={() =>
                  setCurrentQuestion((prev) => Math.max(prev - 1, 0))
                }
                disabled={currentQuestion === 0}
                className="rounded-2xl border border-slate-700 px-5 py-3 font-semibold disabled:opacity-40"
              >
                Previous
              </button>

              <button
                onClick={() =>
                  setCurrentQuestion((prev) =>
                    Math.min(prev + 1, lesson.quiz.questions.length - 1)
                  )
                }
                disabled={currentQuestion === lesson.quiz.questions.length - 1}
                className="rounded-2xl border border-slate-700 px-5 py-3 font-semibold disabled:opacity-40"
              >
                Next
              </button>

              {!submitted ? (
                <button
                  onClick={handleSubmitQuiz}
                  className="rounded-2xl bg-yellow-400 px-5 py-3 font-bold text-slate-950 hover:bg-yellow-300"
                >
                  Submit Test
                </button>
              ) : null}
            </div>
          </div>

          {submitted ? (
            <div className="mt-8 rounded-2xl border border-slate-800 bg-slate-950 p-6">
              <p className="text-sm font-bold uppercase tracking-wide text-yellow-300">
                Results
              </p>
              <h3 className="mt-3 text-3xl font-black">
                {score}% {passed ? "Passed" : "Needs Review"}
              </h3>
              <p className="mt-3 text-slate-300">{lesson.completion.message}</p>

              <div className="mt-5 rounded-2xl border border-slate-800 bg-slate-900 p-5">
                <p className="text-sm font-bold uppercase tracking-wide text-yellow-300">
                  NOVA Completion Voice
                </p>
                <p className="mt-3 text-slate-200">{lesson.completion.novaVoice}</p>
              </div>
            </div>
          ) : null}
        </div>
      </div>
    </div>
  );
}
