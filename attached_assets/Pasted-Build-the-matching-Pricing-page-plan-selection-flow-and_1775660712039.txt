src/lib/accessControl.js
export function canAccessPage(user, pageKey) {
  if (pageKey === "home") return true;
  if (pageKey === "pricing") return true;
  if (pageKey === "choose-plan") return true;
  if (pageKey === "checkout-personal") return true;
  if (pageKey === "checkout-company") return true;

  if (!user) return false;
  if (!user.isSubscribed && user.role !== "owner") return false;

  const { role, subscriptionPlan } = user;

  if (role === "owner") return true;

  if (pageKey === "users-access" || pageKey === "owner") {
    return false;
  }

  if (subscriptionPlan === "personal") {
    return [
      "training",
      "nova-help",
      "nova-trainer",
      "common-mistakes",
      "leaderboard",
      "selector-nation",
    ].includes(pageKey);
  }

  if (subscriptionPlan === "company") {
    if (role === "selector") {
      return [
        "training",
        "nova-help",
        "nova-trainer",
        "common-mistakes",
        "leaderboard",
        "selector-nation",
      ].includes(pageKey);
    }

    if (role === "trainer") {
      return [
        "training",
        "nova-help",
        "nova-trainer",
        "common-mistakes",
        "leaderboard",
        "selector-nation",
        "trainer-dashboard",
      ].includes(pageKey);
    }

    if (role === "supervisor") {
      return [
        "training",
        "nova-help",
        "nova-trainer",
        "common-mistakes",
        "leaderboard",
        "selector-nation",
        "trainer-dashboard",
        "supervisor-dashboard",
      ].includes(pageKey);
    }
  }

  return false;
}

import { useNavigate } from "react-router-dom";
import { useAuthStore } from "../../store/authStore";

export default function PersonalCheckoutPage() {
  const navigate = useNavigate();
  const { user, setUser } = useAuthStore();

  const completeCheckout = () => {
    if (!user) return navigate("/");

    setUser({
      ...user,
      isSubscribed: true,
      subscriptionPlan: "personal",
      role: user.role || "selector",

    navigate("/training");

  return (
    <div className="min-h-screen bg-slate-950 text-white px-6 py-12">

          <p>• Training</p>
          <p>• NOVA Help</p>

function StatCard({ label, value, tone = "default" }) {
  const toneClass =
    tone === "danger" ? "text-red-300" : "text-white";

  return (
    <div className="rounded-3xl border border-slate-800 bg-slate-900 p-5">
      <p className="text-sm text-slate-400">{label}</p>
      <p className={`mt-3 text-2xl font-black ${toneClass}`}>{value}</p>
    </div>
  );
}
