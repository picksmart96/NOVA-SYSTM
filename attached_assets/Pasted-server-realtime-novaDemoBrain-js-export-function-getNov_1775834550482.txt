import { useEffect, useRef, useState } from "react";

export default function NovaVoiceAgent() {
  const recognitionRef = useRef(null);
  const [listening, setListening] = useState(false);

  const speak = (text) => {
    const utter = new SpeechSynthesisUtterance(text);
    utter.rate = 1;
    utter.pitch = 1;

    speechSynthesis.cancel();
    speechSynthesis.speak(utter);
  };

  const startListening = () => {
    const SpeechRecognition =
      window.SpeechRecognition || window.webkitSpeechRecognition;

    const recognition = new SpeechRecognition();
    recognition.continuous = true;
    recognition.interimResults = false;
    recognition.lang = "en-US";

    recognition.onstart = () => setListening(true);

    recognition.onresult = async (event) => {
      const text =
        event.results[event.results.length - 1][0].transcript;

      console.log("User:", text);

      // WAKE WORD CHECK
      if (
        text.toLowerCase().includes("hey nova") ||
        text.toLowerCase().includes("nova")
      ) {
        speak("Yes, how can I help?");
        return;
      }

      // SEND TO AI
      const res = await fetch("/api/nova-help", {
        method: "POST",
        headers: {
          "Content-Type": "application/json",
        },
        body: JSON.stringify({ message: text }),
      });

      const data = await res.json();

      speak(data.reply);
    };

    recognition.onend = () => {
      recognition.start(); // 🔥 KEEP ALIVE
    };

    recognition.start();
    recognitionRef.current = recognition;
  };

  useEffect(() => {
    // 🔓 Unlock audio (mobile)
    const unlock = () => {
      speak(" ");
      document.removeEventListener("click", unlock);
    };

    document.addEventListener("click", unlock);

    // 🎤 START NOVA
    setTimeout(() => {
      speak(
        "Hi, I'm NOVA. I'm here to help you understand how PickSmart improves warehouse training. Just ask me anything."
      );

      startListening();
    }, 1000);
  }, []);

  return (
    <div className="min-h-screen flex flex-col items-center justify-center bg-slate-950 text-white">
      <h1 className="text-4xl font-bold text-yellow-400">NOVA</h1>

      <p className="mt-4 text-slate-300">
        {listening ? "Listening..." : "Starting..."}
      </p>

      <div className="mt-6 animate-pulse text-yellow-400">
        🎤 Voice Active
      </div>
    </div>
  );
}
🧠 3️⃣ BACKEND AI ROUTE
server/routes/novaHelpApi.js
import express from "express";
import OpenAI from "openai";
import { getNovaDemoBrain } from "../realtime/novaDemoBrain.js";

const router = express.Router();

const openai = new OpenAI({
  apiKey: process.env.OPENAI_API_KEY,
});

router.post("/", async (req, res) => {
  const { message } = req.body;

  const response = await openai.chat.completions.create({
    model: "gpt-4o-mini",
    messages: [
      {
        role: "system",
        content: getNovaDemoBrain(),
      },
      {
        role: "user",
        content: message,
      },
    ],
  });

  res.json({
    reply: response.choices[0].message.content,
  });
});

export default router;
