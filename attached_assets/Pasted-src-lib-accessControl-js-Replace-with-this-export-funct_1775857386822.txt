import { useState } from "react";
import { useAuthStore } from "../../store/authStore";
import RequestAccessModal from "./RequestAccessModal";

export default function DemoLockedAction({
  children,
  onAllowedClick,
  className = "",
  title,
}) {
  const { user } = useAuthStore();
  const [open, setOpen] = useState(false);

  const blocked = !!user?.isDemoUser;

  const handleClick = () => {
    if (blocked) {
      setOpen(true);
      return;
    }

    onAllowedClick?.();
  };

  return (
    <>
      <div className={className} onClick={handleClick}>
        {children}
      </div>

      <RequestAccessModal
        open={open}
        onClose={() => setOpen(false)}
        title={title || "Request Access to Unlock This Feature"}
      />
    </>
  );
}

import { useEffect, useMemo, useRef, useState } from "react";

function getRecognitionCtor() {
  return window.SpeechRecognition || window.webkitSpeechRecognition || null;
}

function getVoicesSafe() {
  return window.speechSynthesis?.getVoices?.() || [];
}

function pickVoice(lang = "en-US") {
  const voices = getVoicesSafe();

  const exact = voices.find((v) => v.lang === lang);
  if (exact) return exact;

  const partial = voices.find((v) =>
    v.lang?.toLowerCase().startsWith(lang.slice(0, 2).toLowerCase())
  );
  if (partial) return partial;

  return voices[0] || null;
}

export default function useNovaDemoVoiceAgent() {
  const recognitionRef = useRef(null);
  const interruptRef = useRef(false);
  const sessionIdRef = useRef(
    `demo-agent-${Math.random().toString(36).slice(2)}-${Date.now()}`
  );

  const [language, setLanguage] = useState("en");
  const [isListening, setIsListening] = useState(false);
  const [isSpeaking, setIsSpeaking] = useState(false);
  const [isThinking, setIsThinking] = useState(false);
  const [lastHeard, setLastHeard] = useState("");
  const [lastReply, setLastReply] = useState("");
  const [error, setError] = useState("");
  const [micPermission, setMicPermission] = useState("unknown");
  const [transcript, setTranscript] = useState([]);
  const [showLeadPrompt, setShowLeadPrompt] = useState(false);

  const stateLabel = useMemo(() => {
    if (error) return "Error";
    if (isThinking) return "Thinking";
    if (isSpeaking) return "Speaking";
    if (isListening) return "Listening";
    return "Idle";
  }, [error, isThinking, isSpeaking, isListening]);

  const currentLangCode = language === "es" ? "es-ES" : "en-US";

  const pushTranscript = (role, text) => {
    setTranscript((prev) => [
      ...prev.slice(-19),
      { id: `${Date.now()}-${Math.random()}`, role, text },
    ]);
  };

  const stopSpeaking = () => {
    interruptRef.current = true;
    window.speechSynthesis.cancel();
    setIsSpeaking(false);
  };

  const speak = (text, { onEnd } = {}) => {
    stopSpeaking();
    interruptRef.current = false;

    const utter = new SpeechSynthesisUtterance(text);
    utter.lang = currentLangCode;
    utter.rate = 1;
    utter.pitch = 1;
    utter.volume = 1;

    const voice = pickVoice(currentLangCode);
    if (voice) utter.voice = voice;

    utter.onstart = () => setIsSpeaking(true);

    utter.onend = () => {
      setIsSpeaking(false);
      if (!interruptRef.current) onEnd?.();
    };

    utter.onerror = () => {
      setIsSpeaking(false);
      onEnd?.();
    };

    window.speechSynthesis.speak(utter);
  };

  const sendToAgent = async (text) => {
    setIsThinking(true);

    try {
      const res = await fetch("/api/demo/nova-agent", {
        method: "POST",
        headers: {
          "Content-Type": "application/json",
        },
        body: JSON.stringify({
          sessionId: sessionIdRef.current,
          message: text,
          language,
        }),
      });

      const data = await res.json();
      const reply =
        data.reply ||
        (language === "es"
          ? "¿Quieres que te explique más?"
          : "Would you like me to explain more?");

      setLastReply(reply);
      pushTranscript("assistant", reply);
      setShowLeadPrompt(!!data.handoff);
      setIsThinking(false);

      speak(reply);
    } catch {
      setIsThinking(false);
      setError("Could not reach NOVA.");
    }
  };

  const startRecognition = () => {
    const Recognition = getRecognitionCtor();

    if (!Recognition) {
      setError("Speech recognition is not supported on this browser.");
      return false;
    }

    if (recognitionRef.current) {
      try {
        recognitionRef.current.stop();
      } catch {}
    }

    const rec = new Recognition();
    rec.lang = currentLangCode;
    rec.continuous = true;
    rec.interimResults = false;
    rec.maxAlternatives = 1;

    rec.onstart = () => {
      setIsListening(true);
      setError("");
    };

    rec.onend = () => {
      setIsListening(false);
      if (!interruptRef.current) {
        try {
          rec.start();
        } catch {}
      }
    };

    rec.onerror = (event) => {
      setError(event?.error || "Voice recognition error.");
      setIsListening(false);
    };

    rec.onresult = async (event) => {
      const text =
        event.results?.[event.results.length - 1]?.[0]?.transcript?.trim() || "";

      if (!text) return;

      setLastHeard(text);
      pushTranscript("user", text);

      const lower = text.toLowerCase();

      if (
        lower.includes("stop nova") ||
        lower === "stop" ||
        lower.includes("be quiet") ||
        lower.includes("parar")
      ) {
        stopSpeaking();
        const msg =
          language === "es"
            ? "Está bien. Me quedaré en silencio hasta que me hagas otra pregunta."
            : "Okay. I'll stay quiet until you ask another question.";
        setLastReply(msg);
        pushTranscript("assistant", msg);
        speak(msg);
        return;
      }

      if (
        lower.includes("reset conversation") ||
        lower.includes("start over") ||
        lower.includes("reiniciar conversación")
      ) {
        stopSpeaking();
        setIsThinking(true);

        try {
          const res = await fetch("/api/demo/nova-agent", {
            method: "POST",
            headers: {
              "Content-Type": "application/json",
            },
            body: JSON.stringify({
              sessionId: sessionIdRef.current,
              reset: true,
              language,
            }),
          });

          const data = await res.json();
          const reply = data.reply || "Conversation reset.";

          setTranscript([]);
          setLastReply(reply);
          pushTranscript("assistant", reply);
          setShowLeadPrompt(false);
          setIsThinking(false);

          speak(reply);
        } catch {
          setIsThinking(false);
          setError("Could not reset conversation.");
        }

        return;
      }

      if (lower.includes("speak spanish") || lower.includes("habla español")) {
        setLanguage("es");
        const msg = "Claro. Ahora hablaré en español. ¿Qué te gustaría saber?";
        setLastReply(msg);
        pushTranscript("assistant", msg);
        speak(msg);
        return;
      }

      if (lower.includes("speak english") || lower.includes("english please")) {
        setLanguage("en");
        const msg = "Sure. I'll continue in English. What would you like to know?";
        setLastReply(msg);
        pushTranscript("assistant", msg);
        speak(msg);
        return;
      }

      await sendToAgent(text);
    };

    recognitionRef.current = rec;

    try {
      rec.start();
      return true;
    } catch {
      setError("Could not start voice recognition.");
      return false;
    }
  };

  const initialize = async () => {
    try {
      if (navigator.mediaDevices?.getUserMedia) {
        await navigator.mediaDevices.getUserMedia({ audio: true });
        setMicPermission("granted");
      }
    } catch {
      setMicPermission("denied");
      setError("Microphone permission denied.");
      return false;
    }

    const started = startRecognition();
    if (!started) return false;

    const welcome =
      language === "es"
        ? "Hola, soy NOVA. Estoy aquí para ayudarte con cualquier pregunta sobre PickSmart Academy. Solo pregunta."
        : "Hi there, I'm NOVA. I'm here to assist you with any questions you have about PickSmart Academy. Just ask.";

    setLastReply(welcome);
    pushTranscript("assistant", welcome);
    speak(welcome);

    return true;
  };

  const destroy = () => {
    interruptRef.current = true;
    try {
      recognitionRef.current?.stop();
    } catch {}
    stopSpeaking();
  };

  useEffect(() => {
    const loadVoices = () => getVoicesSafe();
    window.speechSynthesis.onvoiceschanged = loadVoices;
    loadVoices();

    return () => {
      destroy();
      window.speechSynthesis.onvoiceschanged = null;
    };
  }, []);

  return {
    initialize,
    destroy,
    stopSpeaking,
    stateLabel,
    isListening,
    isSpeaking,
    isThinking,
    lastHeard,
    lastReply,
    transcript,
    error,
    micPermission,
    language,
    setLanguage,
    showLeadPrompt,
    setShowLeadPrompt,
  };
}

        <div className="grid gap-6 lg:grid-cols-2">
          <div className="rounded-3xl border border-slate-800 bg-slate-900 p-6">
            <h2 className="text-2xl font-bold">Last Heard</h2>
            <p className="mt-4 min-h-[120px] rounded-2xl border border-slate-800 bg-slate-950 p-4 text-slate-300">
              {agent.lastHeard || "Waiting for your question."}
            </p>
          </div>

          <div className="rounded-3xl border border-slate-800 bg-slate-900 p-6">
            <h2 className="text-2xl font-bold">Last Reply</h2>
            <p className="mt-4 min-h-[120px] rounded-2xl border border-slate-800 bg-slate-950 p-4 text-slate-300">
              {agent.lastReply || "NOVA will reply here."}
            </p>
          </div>
        </div>

        <div className="rounded-3xl border border-slate-800 bg-slate-900 p-6">
          <h2 className="text-2xl font-bold">Conversation Transcript</h2>
          <div className="mt-5 max-h-[420px] space-y-3 overflow-auto">
            {agent.transcript.length === 0 ? (
              <div className="rounded-2xl border border-slate-800 bg-slate-950 p-4 text-slate-400">
                The conversation transcript will appear here.
              </div>
            ) : (
              agent.transcript.map((item) => (
                <div
                  key={item.id}
                  className={`rounded-2xl border p-4 ${
                    item.role === "assistant"
                      ? "border-yellow-400/30 bg-yellow-500/10 text-yellow-100"
                      : "border-slate-800 bg-slate-950 text-slate-300"
                  }`}
                >
                  <p className="text-xs font-bold uppercase tracking-wide opacity-70">
                    {item.role === "assistant" ? "NOVA" : "You"}
                  </p>
                  <p className="mt-2">{item.text}</p>
                </div>
              ))
            )}
          </div>
        </div>

        {agent.showLeadPrompt ? (
          <div className="rounded-3xl border border-yellow-400 bg-yellow-500/10 p-6 text-center">
            <h3 className="text-2xl font-bold text-yellow-200">
              Want to see how this works for your team?
            </h3>
            <p className="mx-auto mt-3 max-w-2xl text-slate-200">
              Request company access and we’ll show you how PickSmart Academy can train, coach, and improve performance for your warehouse team.
            </p>

            <div className="mt-6 flex flex-wrap justify-center gap-3">
              <button
                onClick={() => navigate("/request-access")}
                className="rounded-2xl bg-yellow-400 px-5 py-3 font-bold text-slate-950 hover:bg-yellow-300"
              >
                Request Company Access
              </button>

              <button
                onClick={() => agent.setShowLeadPrompt(false)}
                className="rounded-2xl border border-slate-700 px-5 py-3 font-semibold hover:border-yellow-400"
              >
                Keep Exploring
              </button>
            </div>
          </div>
        ) : null}

        {agent.error ? (
          <div className="rounded-3xl border border-red-500/30 bg-red-500/10 p-5 text-red-200">
            {agent.error}
          </div>
        ) : null}
      </div>
    </div>
  );
}
