import { useEffect, useMemo, useRef, useState } from "react";
import { assignments } from "../../data/assignments";
import { assignmentStops } from "../../data/assignmentStops";
import { voiceCommands } from "../../data/voiceCommands";
import {
  normalizeSpeech,
  isConfirm,
  isDeny,
  isLoadPicks,
  extractDigits,
} from "../../lib/parser";

const SYSTEM_DEFAULTS = {
  printerNumber: "307",
  alphaLabelNumber: "242",
  bravoLabelNumber: "578",
};

const SAFETY_ITEMS = [
  "Brakes okay?",
  "Battery guard okay?",
  "Horn okay?",
  "Wheels okay?",
  "Hydraulics okay?",
  "Controls okay?",
  "Steering okay?",
  "Welds okay?",
  "Electric wiring okay?",
];

const DEMO_SELECTOR_USER_ID = "user-001";

function speakText(text, onEnd) {
  if (!("speechSynthesis" in window)) {
    onEnd?.();
    return;
  }

  window.speechSynthesis.cancel();

  const utterance = new SpeechSynthesisUtterance(text);
  utterance.rate = 1;
  utterance.pitch = 1;

  const voices = window.speechSynthesis.getVoices();
  const preferredVoice =
    voices.find((v) => /samantha|zira|karen|female/i.test(v.name)) || voices[0];

  if (preferredVoice) utterance.voice = preferredVoice;

  utterance.onend = () => onEnd?.();
  window.speechSynthesis.speak(utterance);
}

export default function NovaTrainerPage() {
  const recognitionRef = useRef(null);
  const shouldKeepListeningRef = useRef(false);

  const [selectorUserId, setSelectorUserId] = useState(DEMO_SELECTOR_USER_ID);

  const [phase, setPhase] = useState("SIGN_ON");
  const [promptText, setPromptText] = useState("Enter equipment ID.");
  const [heardText, setHeardText] = useState("");
  const [commandLog, setCommandLog] = useState([]);
  const [listening, setListening] = useState(false);
  const [speaking, setSpeaking] = useState(false);
  const [thinking, setThinking] = useState(false);
  const [errorText, setErrorText] = useState("");

  const [equipmentId, setEquipmentId] = useState("");
  const [maxPalletCount, setMaxPalletCount] = useState("2");
  const [safetyIndex, setSafetyIndex] = useState(0);
  const [failedSafetyItem, setFailedSafetyItem] = useState("");

  const [activeAssignmentId, setActiveAssignmentId] = useState(null);
  const [currentStopIndex, setCurrentStopIndex] = useState(0);
  const [invalidCount, setInvalidCount] = useState(0);

  const [alphaReady, setAlphaReady] = useState(false);
  const [bravoReady, setBravoReady] = useState(false);

  const [batchStage, setBatchStage] = useState("NOT_STARTED");
  const [completionStep, setCompletionStep] = useState(0);

  const activeAssignment = useMemo(() => {
    return assignments.find((a) => a.id === activeAssignmentId) || null;
  }, [activeAssignmentId]);

  const stops = useMemo(() => {
    if (!activeAssignment) return [];
    return assignmentStops[activeAssignment.assignmentNumber] || [];
  }, [activeAssignment]);

  const currentStop = stops[currentStopIndex] || null;
  const nextStop = stops[currentStopIndex + 1] || null;

  const myAssignments = useMemo(() => {
    return assignments.filter(
      (a) => !a.archived && a.selectorUserId === selectorUserId
    );
  }, [selectorUserId]);

  const activeCommandList = useMemo(() => {
    return voiceCommands;
  }, []);

  const addLog = (type, text) => {
    setCommandLog((prev) => [
      {
        id: Date.now() + Math.random(),
        type,
        text,
        time: new Date().toLocaleTimeString(),
      },
      ...prev,
    ]);
  };

  const setPromptAndSpeak = (text) => {
    setPromptText(text);
    setSpeaking(true);
    addLog("NOVA", text);
    speakText(text, () => {
      setSpeaking(false);
    });
  };

  const startListening = () => {
    const SpeechRecognition =
      window.SpeechRecognition || window.webkitSpeechRecognition;

    if (!SpeechRecognition) {
      setErrorText("Speech recognition is not supported in this browser.");
      return;
    }

    if (!recognitionRef.current) {
      const recognition = new SpeechRecognition();
      recognition.continuous = false;
      recognition.interimResults = false;
      recognition.lang = "en-US";

      recognition.onstart = () => {
        setListening(true);
        setThinking(false);
        setErrorText("");
      };

      recognition.onend = () => {
        setListening(false);
        if (shouldKeepListeningRef.current && !speaking) {
          setTimeout(() => {
            try {
              recognition.start();
            } catch {}
          }, 250);
        }
      };

      recognition.onerror = () => {
        setErrorText("Speech recognition error.");
        setListening(false);
        if (shouldKeepListeningRef.current && !speaking) {
          setTimeout(() => {
            try {
              recognition.start();
            } catch {}
          }, 400);
        }
      };

      recognition.onresult = (event) => {
        const transcript = normalizeSpeech(
          event.results?.[0]?.[0]?.transcript || ""
        );
        setHeardText(transcript);
        setThinking(false);
        addLog("USER", transcript || "[no input]");
        handleVoiceInput(transcript);
      };

      recognitionRef.current = recognition;
    }

    shouldKeepListeningRef.current = true;
    setThinking(true);

    try {
      recognitionRef.current.start();
    } catch {}
  };

  const stopListening = () => {
    shouldKeepListeningRef.current = false;
    setListening(false);
    setThinking(false);
    try {
      recognitionRef.current?.stop();
    } catch {}
  };

  useEffect(() => {
    if (!speaking && shouldKeepListeningRef.current && !listening) {
      const t = setTimeout(() => startListening(), 350);
      return () => clearTimeout(t);
    }
  }, [speaking, listening]);

  useEffect(() => {
    setPromptAndSpeak("Enter equipment ID.");
    startListening();
    // eslint-disable-next-line react-hooks/exhaustive-deps
  }, []);

  const handleCommandShortcuts = (input) => {
    const normalized = normalizeSpeech(input);

    if (!normalized) {
      setPromptAndSpeak("No input received.");
      return true;
    }

    if (normalized === "repeat labels") {
      setPromptAndSpeak("Repeating label instructions.");
      return true;
    }

    if (normalized === "makeup skips") {
      setPromptAndSpeak("Makeup skips requested.");
      return true;
    }

    if (normalized === "get drops") {
      setPromptAndSpeak("Get drops requested.");
      return true;
    }

    if (normalized === "close batch") {
      setPromptAndSpeak("Closing batch.");
      setPhase("BATCH_COMPLETE");
      return true;
    }

    if (normalized.startsWith("go to aisle")) {
      setPromptAndSpeak("Go to aisle command received.");
      return true;
    }

    if (normalized === "base items") {
      setPromptAndSpeak("Base items review not yet loaded.");
      return true;
    }

    if (normalized === "previous") {
      setPromptAndSpeak("Previous item requested.");
      return true;
    }

    if (normalized === "base complete") {
      setPromptAndSpeak("Base complete confirmed.");
      return true;
    }

    if (normalized === "remove pallet") {
      setPromptAndSpeak("Remove pallet noted.");
      return true;
    }

    if (normalized === "skip slot") {
      setPromptAndSpeak("Skip slot noted.");
      return true;
    }

    if (normalized === "report damage") {
      setPromptAndSpeak("Damage flow started.");
      return true;
    }

    if (normalized.startsWith("damage")) {
      setPromptAndSpeak("Damage quantity recorded.");
      return true;
    }

    if (normalized === "slot empty") {
      setPromptAndSpeak("Slot empty recorded.");
      return true;
    }

    if (normalized === "recap weights") {
      setPromptAndSpeak("Recapping weights.");
      return true;
    }

    if (normalized === "next") {
      setPromptAndSpeak("Next item.");
      return true;
    }

    if (normalized === "exit list") {
      setPromptAndSpeak("Exit list.");
      return true;
    }

    if (normalized === "no labels") {
      setPromptAndSpeak("No labels recorded.");
      return true;
    }

    return false;
  };

  const startAssignedBatch = () => {
    const assignment = myAssignments[0];
    if (!assignment) {
      setPromptAndSpeak("No assignment assigned to your ID.");
      return;
    }

    setActiveAssignmentId(assignment.id);
    setCurrentStopIndex(0);
    setBatchStage("SUMMARY");

    const summary = `Start aisle ${assignment.startAisle} end aisle ${assignment.endAisle}. Slots ${assignment.stops}. Total cases ${assignment.totalCases}. Pallets ${assignment.pallets}. Estimated goal ${assignment.goalTimeMinutes}. To continue say ready.`;
    setPhase("LOAD_PICKS_SUMMARY");
    setPromptAndSpeak(summary);
  };

  const beginPickFlow = () => {
    if (!currentStop) return;
    setPhase("PICKING");
    setPromptAndSpeak(`New aisle ${currentStop.aisle} slot ${currentStop.slot}`);
  };

  const moveToNextStop = () => {
    if (currentStopIndex < stops.length - 1) {
      const nextIndex = currentStopIndex + 1;
      setCurrentStopIndex(nextIndex);
      const next = stops[nextIndex];
      setPromptAndSpeak(`New aisle ${next.aisle} slot ${next.slot}`);
      return;
    }

    setPhase("BATCH_COMPLETE");
    setCompletionStep(0);
    setPromptAndSpeak(
      `Batch complete deliver pallet bravo to door ${activeAssignment.doorNumber}`
    );
  };

  const handleSignOnFlow = (input) => {
    if (!equipmentId) {
      const digits = extractDigits(input);
      if (!digits) {
        setPromptAndSpeak("Enter equipment ID.");
        return;
      }

      setEquipmentId(digits);
      setPromptAndSpeak(`Confirm equipment ${digits}.`);
      return;
    }

    if (equipmentId && phase === "SIGN_ON") {
      if (isConfirm(input)) {
        setPhase("PALLET_COUNT");
        setPromptAndSpeak(`Enter maximum pallet count for jack ${equipmentId}.`);
        return;
      }

      if (isDeny(input)) {
        setEquipmentId("");
        setPromptAndSpeak("Enter equipment ID.");
        return;
      }

      setPromptAndSpeak(`Confirm equipment ${equipmentId}.`);
    }
  };

  const handleVoiceInput = (rawInput) => {
    const input = normalizeSpeech(rawInput);

    if (handleCommandShortcuts(input)) return;

    if (phase === "SIGN_ON") {
      handleSignOnFlow(input);
      return;
    }

    if (phase === "PALLET_COUNT") {
      const digits = extractDigits(input);
      if (!digits) {
        setPromptAndSpeak("Enter maximum pallet count.");
        return;
      }

      setMaxPalletCount(digits);
      setPhase("PALLET_COUNT_CONFIRM");
      setPromptAndSpeak("Confirm maximum pallet count.");
      return;
    }

    if (phase === "PALLET_COUNT_CONFIRM") {
      if (isConfirm(input)) {
        setPhase("SAFETY");
        setSafetyIndex(0);
        setPromptAndSpeak(SAFETY_ITEMS[0]);
        return;
      }

      if (isDeny(input)) {
        setPhase("PALLET_COUNT");
        setPromptAndSpeak(`Enter maximum pallet count for jack ${equipmentId}.`);
        return;
      }

      setPromptAndSpeak("Confirm maximum pallet count.");
      return;
    }

    if (phase === "SAFETY") {
      const currentSafety = SAFETY_ITEMS[safetyIndex];

      if (isConfirm(input)) {
        if (safetyIndex < SAFETY_ITEMS.length - 1) {
          const nextIndex = safetyIndex + 1;
          setSafetyIndex(nextIndex);
          setPromptAndSpeak(SAFETY_ITEMS[nextIndex]);
        } else {
          setPhase("LOAD_PICKS_WAIT");
          setPromptAndSpeak("Say load picks.");
        }
        return;
      }

      if (isDeny(input)) {
        setFailedSafetyItem(currentSafety);
        setPhase("SAFETY_FAILED");
        setPromptAndSpeak(`Safety failed. ${currentSafety} Session stopped.`);
        stopListening();
        return;
      }

      setPromptAndSpeak(currentSafety);
      return;
    }

    if (phase === "LOAD_PICKS_WAIT") {
      if (isLoadPicks(input)) {
        startAssignedBatch();
        return;
      }

      setPromptAndSpeak("Say load picks.");
      return;
    }

    if (phase === "LOAD_PICKS_SUMMARY") {
      if (input === "ready") {
        setPhase("ALPHA_SETUP");
        setPromptAndSpeak("Position alpha pallet get chep.");
        return;
      }

      setPromptAndSpeak("To continue say ready.");
      return;
    }

    if (phase === "ALPHA_SETUP") {
      if (input === "ready") {
        setAlphaReady(true);
        setPhase("BRAVO_SETUP");
        setPromptAndSpeak("Position bravo pallet get chep.");
        return;
      }

      setPromptAndSpeak("Position alpha pallet get chep.");
      return;
    }

    if (phase === "BRAVO_SETUP") {
      if (input === "ready") {
        setBravoReady(true);
        beginPickFlow();
        return;
      }

      setPromptAndSpeak("Position bravo pallet get chep.");
      return;
    }

    if (phase === "PICKING") {
      if (!currentStop) return;

      const digits = extractDigits(input);

      if (!digits) {
        setInvalidCount((c) => c + 1);
        setPromptAndSpeak(`You said ${input || "nothing"}. Invalid.`);
        return;
      }

      if (digits !== currentStop.checkCode) {
        setInvalidCount((c) => c + 1);
        setPromptAndSpeak(`You said ${digits}. Invalid.`);
        return;
      }

      setPhase("PICK_QTY_CONFIRM");
      const palletSide = currentStopIndex % 2 === 0 ? "Alpha" : "Bravo";
      setPromptAndSpeak(`Grab ${currentStop.qty} ${palletSide}`);
      return;
    }

    if (phase === "PICK_QTY_CONFIRM") {
      if (input === "ready" || input === "grab" || isConfirm(input)) {
        moveToNextStop();
        return;
      }

      setPromptAndSpeak("Ready.");
      return;
    }

    if (phase === "BATCH_COMPLETE") {
      if (!activeAssignment) return;

      const doorCode = String(activeAssignment.doorCode);

      if (completionStep === 0) {
        const digits = extractDigits(input);
        if (digits === doorCode) {
          setCompletionStep(1);
          setPromptAndSpeak("Apply labels to pallet alpha");
          return;
        }

        setPromptAndSpeak(
          `Batch complete deliver pallet bravo to door ${activeAssignment.doorNumber}`
        );
        return;
      }

      if (completionStep === 1) {
        if (
          input.includes("alpha") ||
          extractDigits(input) === SYSTEM_DEFAULTS.alphaLabelNumber
        ) {
          setCompletionStep(2);
          setPromptAndSpeak("Apply labels to pallet bravo");
          return;
        }

        setPromptAndSpeak("Apply labels to pallet alpha");
        return;
      }

      if (completionStep === 2) {
        if (
          input.includes("bravo") ||
          extractDigits(input) === SYSTEM_DEFAULTS.bravoLabelNumber
        ) {
          setCompletionStep(3);
          setPromptAndSpeak(
            `Deliver bravo pallet to door ${activeAssignment.doorNumber}`
          );
          return;
        }

        setPromptAndSpeak("Apply labels to pallet bravo");
        return;
      }

      if (completionStep === 3) {
        const digits = extractDigits(input);
        if (digits === doorCode) {
          setCompletionStep(4);
          setPromptAndSpeak(
            `Deliver alpha pallet to door ${activeAssignment.doorNumber}`
          );
          return;
        }

        setPromptAndSpeak(
          `Deliver bravo pallet to door ${activeAssignment.doorNumber}`
        );
        return;
      }

      if (completionStep === 4) {
        const digits = extractDigits(input);
        if (digits === doorCode) {
          setCompletionStep(5);
          setPhase("NEXT_BATCH_WAIT");
          setPromptAndSpeak("Say load picks.");
          return;
        }

        setPromptAndSpeak(
          `Deliver alpha pallet to door ${activeAssignment.doorNumber}`
        );
        return;
      }

      return;
    }

    if (phase === "NEXT_BATCH_WAIT") {
      if (isLoadPicks(input)) {
        startAssignedBatch();
        return;
      }

      setPromptAndSpeak("Say load picks.");
    }
  };

  const manualSay = () => {
    startListening();
  };

  const resetSession = () => {
    stopListening();
    setPhase("SIGN_ON");
    setPromptText("Enter equipment ID.");
    setHeardText("");
    setCommandLog([]);
    setErrorText("");
    setEquipmentId("");
    setMaxPalletCount("2");
    setSafetyIndex(0);
    setFailedSafetyItem("");
    setActiveAssignmentId(null);
    setCurrentStopIndex(0);
    setInvalidCount(0);
    setAlphaReady(false);
    setBravoReady(false);
    setBatchStage("NOT_STARTED");
    setCompletionStep(0);

    setTimeout(() => {
      setPromptAndSpeak("Enter equipment ID.");
      startListening();
    }, 300);
  };

  const phaseLabelMap = {
    SIGN_ON: "Sign On",
    PALLET_COUNT: "Pallet Count",
    PALLET_COUNT_CONFIRM: "Pallet Confirm",
    SAFETY: "Safety",
    SAFETY_FAILED: "Safety Failed",
    LOAD_PICKS_WAIT: "Load Picks",
    LOAD_PICKS_SUMMARY: "Load Summary",
    ALPHA_SETUP: "Alpha Setup",
    BRAVO_SETUP: "Bravo Setup",
    PICKING: "Picking",
    PICK_QTY_CONFIRM: "Pick Confirm",
    BATCH_COMPLETE: "Batch Complete",
    NEXT_BATCH_WAIT: "Next Batch",
  };

  return (
    <div className="min-h-screen bg-slate-950 text-white px-6 py-8">
      <div className="max-w-7xl mx-auto space-y-8">
        {/* Header */}
        <div className="flex flex-col xl:flex-row xl:items-start xl:justify-between gap-4">
          <div>
            <p className="text-yellow-400 text-sm font-semibold uppercase tracking-[0.2em]">
              NOVA Trainer
            </p>
            <h1 className="mt-2 text-4xl font-black">ES3 Script Mode</h1>
            <p className="mt-2 text-slate-400">
              Jennifer-style sign on, safety check, load picks, picking, and batch complete flow.
            </p>
          </div>

          <div className="flex flex-wrap gap-3">
            <button
              onClick={manualSay}
              className="rounded-2xl border border-slate-700 bg-slate-900 px-5 py-3 font-semibold hover:border-yellow-400 transition"
            >
              Listen Now
            </button>
            <button
              onClick={resetSession}
              className="rounded-2xl bg-yellow-400 px-5 py-3 font-bold text-slate-950 hover:bg-yellow-300 transition"
            >
              Reset Session
            </button>
          </div>
        </div>

        {/* Top stats */}
        <div className="grid sm:grid-cols-2 xl:grid-cols-6 gap-4">
          <div className="rounded-3xl border border-slate-800 bg-slate-900 p-5">
            <p className="text-sm text-slate-400">Phase</p>
            <p className="mt-3 text-2xl font-black">{phaseLabelMap[phase]}</p>
          </div>

          <div className="rounded-3xl border border-slate-800 bg-slate-900 p-5">
            <p className="text-sm text-slate-400">Selector</p>
            <p className="mt-3 text-2xl font-black">{selectorUserId}</p>
          </div>

          <div className="rounded-3xl border border-slate-800 bg-slate-900 p-5">
            <p className="text-sm text-slate-400">Assignment</p>
            <p className="mt-3 text-2xl font-black">
              {activeAssignment ? `#${activeAssignment.assignmentNumber}` : "—"}
            </p>
          </div>

          <div className="rounded-3xl border border-slate-800 bg-slate-900 p-5">
            <p className="text-sm text-slate-400">Current Stop</p>
            <p className="mt-3 text-2xl font-black">
              {currentStop ? currentStop.stopOrder : "—"}
            </p>
          </div>

          <div className="rounded-3xl border border-slate-800 bg-slate-900 p-5">
            <p className="text-sm text-slate-400">Listening</p>
            <p className="mt-3 text-2xl font-black">
              {listening ? "ON" : "OFF"}
            </p>
          </div>

          <div className="rounded-3xl border border-slate-800 bg-slate-900 p-5">
            <p className="text-sm text-slate-400">Invalid Count</p>
            <p className="mt-3 text-2xl font-black">{invalidCount}</p>
          </div>
        </div>

        {/* Main 2-column */}
        <div className="grid xl:grid-cols-[1.25fr_0.75fr] gap-6">
          {/* Left main panel */}
          <div className="space-y-6">
            <div className="rounded-3xl border border-slate-800 bg-slate-900 p-6 shadow-lg">
              <h2 className="text-2xl font-bold">Current Prompt</h2>
              <div className="mt-5 rounded-2xl border border-slate-800 bg-slate-950 p-5">
                <p className="text-lg text-white font-semibold">{promptText}</p>
              </div>

              <div className="mt-6 grid md:grid-cols-3 gap-4">
                <div className="rounded-2xl border border-slate-800 bg-slate-950 p-4">
                  <p className="text-sm text-slate-400">Heard Response</p>
                  <p className="mt-2 font-semibold">{heardText || "—"}</p>
                </div>

                <div className="rounded-2xl border border-slate-800 bg-slate-950 p-4">
                  <p className="text-sm text-slate-400">Voice State</p>
                  <p className="mt-2 font-semibold">
                    {speaking
                      ? "Speaking"
                      : thinking
                      ? "Thinking"
                      : listening
                      ? "Listening"
                      : "Idle"}
                  </p>
                </div>

                <div className="rounded-2xl border border-slate-800 bg-slate-950 p-4">
                  <p className="text-sm text-slate-400">Error</p>
                  <p className="mt-2 font-semibold text-red-300">
                    {errorText || "None"}
                  </p>
                </div>
              </div>
            </div>

            <div className="rounded-3xl border border-slate-800 bg-slate-900 p-6 shadow-lg">
              <h2 className="text-2xl font-bold">Session Detail</h2>

              <div className="mt-6 grid md:grid-cols-2 gap-4">
                <div className="rounded-2xl border border-slate-800 bg-slate-950 p-4">
                  <p className="text-sm text-slate-400">Equipment ID</p>
                  <p className="mt-2 font-semibold">{equipmentId || "—"}</p>
                </div>

                <div className="rounded-2xl border border-slate-800 bg-slate-950 p-4">
                  <p className="text-sm text-slate-400">Max Pallet Count</p>
                  <p className="mt-2 font-semibold">{maxPalletCount || "—"}</p>
                </div>

                <div className="rounded-2xl border border-slate-800 bg-slate-950 p-4">
                  <p className="text-sm text-slate-400">Current Aisle</p>
                  <p className="mt-2 font-semibold">{currentStop?.aisle ?? "—"}</p>
                </div>

                <div className="rounded-2xl border border-slate-800 bg-slate-950 p-4">
                  <p className="text-sm text-slate-400">Current Slot</p>
                  <p className="mt-2 font-semibold">{currentStop?.slot ?? "—"}</p>
                </div>

                <div className="rounded-2xl border border-slate-800 bg-slate-950 p-4">
                  <p className="text-sm text-slate-400">Next Stop</p>
                  <p className="mt-2 font-semibold">
                    {nextStop ? `${nextStop.aisle}/${nextStop.slot}` : "—"}
                  </p>
                </div>

                <div className="rounded-2xl border border-slate-800 bg-slate-950 p-4">
                  <p className="text-sm text-slate-400">Safety Status</p>
                  <p className="mt-2 font-semibold">
                    {failedSafetyItem
                      ? `Failed: ${failedSafetyItem}`
                      : phase === "SAFETY" || phase === "SAFETY_FAILED"
                      ? `Item ${safetyIndex + 1}/${SAFETY_ITEMS.length}`
                      : "Passed / In Progress"}
                  </p>
                </div>

                <div className="rounded-2xl border border-slate-800 bg-slate-950 p-4">
                  <p className="text-sm text-slate-400">Door</p>
                  <p className="mt-2 font-semibold">
                    {activeAssignment?.doorNumber ?? "—"}
                  </p>
                </div>

                <div className="rounded-2xl border border-slate-800 bg-slate-950 p-4">
                  <p className="text-sm text-slate-400">Door Code</p>
                  <p className="mt-2 font-semibold">
                    {activeAssignment?.doorCode ?? "—"}
                  </p>
                </div>

                <div className="rounded-2xl border border-slate-800 bg-slate-950 p-4">
                  <p className="text-sm text-slate-400">Alpha Label</p>
                  <p className="mt-2 font-semibold">
                    {SYSTEM_DEFAULTS.alphaLabelNumber}
                  </p>
                </div>

                <div className="rounded-2xl border border-slate-800 bg-slate-950 p-4">
                  <p className="text-sm text-slate-400">Bravo Label</p>
                  <p className="mt-2 font-semibold">
                    {SYSTEM_DEFAULTS.bravoLabelNumber}
                  </p>
                </div>
              </div>
            </div>

            <div className="rounded-3xl border border-slate-800 bg-slate-900 p-6 shadow-lg">
              <h2 className="text-2xl font-bold">Command Log</h2>

              <div className="mt-5 max-h-[360px] overflow-auto space-y-3">
                {commandLog.length === 0 ? (
                  <div className="rounded-2xl border border-slate-800 bg-slate-950 p-4 text-slate-400">
                    No commands logged yet.
                  </div>
                ) : (
                  commandLog.map((entry) => (
                    <div
                      key={entry.id}
                      className="rounded-2xl border border-slate-800 bg-slate-950 p-4"
                    >
                      <div className="flex items-center justify-between gap-3">
                        <span
                          className={`rounded-full px-3 py-1 text-xs font-bold ${
                            entry.type === "NOVA"
                              ? "bg-yellow-500/20 text-yellow-300"
                              : "bg-blue-500/20 text-blue-300"
                          }`}
                        >
                          {entry.type}
                        </span>
                        <span className="text-xs text-slate-500">
                          {entry.time}
                        </span>
                      </div>
                      <p className="mt-3 text-sm text-slate-200">{entry.text}</p>
                    </div>
                  ))
                )}
              </div>
            </div>
          </div>

          {/* Right side panel */}
          <div className="space-y-6">
            <div className="rounded-3xl border border-slate-800 bg-slate-900 p-6 shadow-lg">
              <h2 className="text-2xl font-bold">ES3 Command Cheat Sheet</h2>
              <div className="mt-5 max-h-[600px] overflow-auto space-y-3">
                {activeCommandList.map((cmd) => (
                  <div
                    key={`${cmd.type}-${cmd.phrase}`}
                    className="rounded-2xl border border-slate-800 bg-slate-950 p-4"
                  >
                    <div className="flex flex-wrap items-center gap-2">
                      <span
                        className={`rounded-full px-3 py-1 text-xs font-bold ${
                          cmd.type === "help"
                            ? "bg-blue-500/20 text-blue-300"
                            : "bg-yellow-500/20 text-yellow-300"
                        }`}
                      >
                        {cmd.type}
                      </span>
                      <span className="font-semibold">{cmd.phrase}</span>
                    </div>
                    <p className="mt-2 text-sm text-slate-400">{cmd.description}</p>
                  </div>
                ))}
              </div>
            </div>

            <div className="rounded-3xl border border-slate-800 bg-slate-900 p-6 shadow-lg">
              <h2 className="text-2xl font-bold">Assigned Batches</h2>
              <div className="mt-5 space-y-3">
                {myAssignments.length === 0 ? (
                  <div className="rounded-2xl border border-slate-800 bg-slate-950 p-4 text-slate-400">
                    No assignments for {selectorUserId}.
                  </div>
                ) : (
                  myAssignments.map((a) => (
                    <div
                      key={a.id}
                      className={`rounded-2xl border p-4 ${
                        activeAssignmentId === a.id
                          ? "border-yellow-400 bg-yellow-500/10"
                          : "border-slate-800 bg-slate-950"
                      }`}
                    >
                      <p className="font-bold">#{a.assignmentNumber}</p>
                      <p className="mt-2 text-sm text-slate-400">
                        {a.totalCases} cases · Aisles {a.startAisle}-{a.endAisle}
                      </p>
                    </div>
                  ))
                )}
              </div>
            </div>

            <div className="rounded-3xl border border-slate-800 bg-slate-900 p-6 shadow-lg">
              <h2 className="text-2xl font-bold">Manual Controls</h2>
              <div className="mt-5 grid grid-cols-2 gap-3">
                <button
                  onClick={() => handleVoiceInput("yes")}
                  className="rounded-2xl border border-slate-700 px-4 py-3 font-semibold hover:border-yellow-400 transition"
                >
                  Confirm
                </button>
                <button
                  onClick={() => handleVoiceInput("no")}
                  className="rounded-2xl border border-slate-700 px-4 py-3 font-semibold hover:border-red-400 transition"
                >
                  Deny
                </button>
                <button
                  onClick={() => handleVoiceInput("load picks")}
                  className="rounded-2xl border border-slate-700 px-4 py-3 font-semibold hover:border-yellow-400 transition"
                >
                  Load Picks
                </button>
                <button
                  onClick={() => handleVoiceInput("ready")}
                  className="rounded-2xl border border-slate-700 px-4 py-3 font-semibold hover:border-yellow-400 transition"
                >
                  Ready
                </button>
              </div>
            </div>
          </div>
        </div>
      </div>
    </div>
  );
}
