import { useState } from "react";
import { useNavigate } from "react-router-dom";
import { useTrainerStore } from "../../store/trainerStore";
import AssignAssignmentModal from "../../components/nova/AssignAssignmentModal";
import ActivateNovaModal from "../../components/nova/ActivateNovaModal";
import LogSessionModal from "../../components/nova/LogSessionModal";
import SessionCard from "../../components/nova/SessionCard";

export default function TrainerDashboardPage() {
  const navigate = useNavigate();
  const { trainer, selectors, sessions, assignments, addSelector } = useTrainerStore();

  const [showAssignModal, setShowAssignModal] = useState(false);
  const [showNovaModal, setShowNovaModal] = useState(false);
  const [showLogModal, setShowLogModal] = useState(false);

  const [form, setForm] = useState({
    fullName: "John Smith",
    age: "22",
    level: "Beginner",
    notes: "",
  });

  const handleRegister = (e) => {
    e.preventDefault();
    if (!form.fullName.trim()) return;

    addSelector({
      name: form.fullName,
      age: Number(form.age),
      level: form.level,
      notes: form.notes,
    });

    setForm({
      fullName: "",
      age: "",
      level: "Beginner",
      notes: "",
    });
  };

  const openAssignmentsCount = selectors.filter(
    (selector) => selector.assignedAssignmentId
  ).length;

  const getAssignmentNumber = (assignmentId) => {
    const found = assignments.find((a) => a.id === assignmentId);
    return found ? found.assignmentNumber : null;
  };

  return (
    <div className="min-h-screen bg-slate-950 text-white px-6 py-8">
      <div className="max-w-7xl mx-auto space-y-8">
        {/* Header */}
        <div className="flex flex-col md:flex-row md:items-start md:justify-between gap-4">
          <div>
            <h1 className="text-4xl font-black">Trainer Dashboard</h1>
            <p className="mt-2 text-slate-400">
              {trainer.name} · {trainer.role}
            </p>
          </div>

          <div className="flex gap-3 flex-wrap">
            <button
              onClick={() => navigate("/warehouse-reference")}
              className="rounded-2xl border border-slate-700 bg-slate-900 px-5 py-3 font-semibold hover:border-yellow-400 transition"
            >
              Warehouse Ref
            </button>
            <button className="rounded-2xl border border-slate-700 bg-slate-900 px-5 py-3 font-semibold hover:border-red-400 transition">
              Sign out
            </button>
          </div>
        </div>

        {/* Stats */}
        <div className="grid sm:grid-cols-3 gap-4">
          <div className="rounded-3xl border border-slate-800 bg-slate-900 p-6 shadow-lg">
            <p className="text-slate-400 text-sm">My Selectors</p>
            <p className="mt-3 text-4xl font-black">{selectors.length}</p>
          </div>

          <div className="rounded-3xl border border-slate-800 bg-slate-900 p-6 shadow-lg">
            <p className="text-slate-400 text-sm">Sessions Logged</p>
            <p className="mt-3 text-4xl font-black">{sessions.length}</p>
          </div>

          <div className="rounded-3xl border border-slate-800 bg-slate-900 p-6 shadow-lg">
            <p className="text-slate-400 text-sm">Open</p>
            <p className="mt-3 text-4xl font-black">{openAssignmentsCount}</p>
          </div>
        </div>

        {/* Action buttons */}
        <div className="flex flex-wrap gap-3">
          <button
            onClick={() => window.scrollTo({ top: 420, behavior: "smooth" })}
            className="rounded-2xl bg-yellow-400 px-5 py-3 font-bold text-slate-950 hover:bg-yellow-300 transition"
          >
            Register Selector
          </button>
          <button
            onClick={() => setShowNovaModal(true)}
            className="rounded-2xl border border-slate-700 bg-slate-900 px-5 py-3 font-semibold hover:border-yellow-400 transition"
          >
            Activate NOVA
          </button>
          <button
            onClick={() => setShowLogModal(true)}
            className="rounded-2xl border border-slate-700 bg-slate-900 px-5 py-3 font-semibold hover:border-yellow-400 transition"
          >
            Log Session
          </button>
          <button
            onClick={() => setShowAssignModal(true)}
            className="rounded-2xl border border-slate-700 bg-slate-900 px-5 py-3 font-semibold hover:border-yellow-400 transition"
          >
            Assign Assignment
          </button>
        </div>

        {/* Main layout */}
        <div className="grid lg:grid-cols-2 gap-6">
          {/* Register Selector */}
          <div className="rounded-3xl border border-slate-800 bg-slate-900 p-6 shadow-lg">
            <h2 className="text-2xl font-bold mb-6">Register Selector</h2>

            <form onSubmit={handleRegister} className="space-y-5">
              <div>
                <label className="block text-sm text-slate-400 mb-2">Full name</label>
                <input
                  value={form.fullName}
                  onChange={(e) =>
                    setForm((prev) => ({ ...prev, fullName: e.target.value }))
                  }
                  className="w-full rounded-2xl border border-slate-700 bg-slate-950 px-4 py-3 outline-none focus:border-yellow-400"
                />
              </div>

              <div>
                <label className="block text-sm text-slate-400 mb-2">Age</label>
                <input
                  value={form.age}
                  onChange={(e) =>
                    setForm((prev) => ({ ...prev, age: e.target.value }))
                  }
                  className="w-full rounded-2xl border border-slate-700 bg-slate-950 px-4 py-3 outline-none focus:border-yellow-400"
                />
              </div>

              <div>
                <label className="block text-sm text-slate-400 mb-2">Level</label>
                <select
                  value={form.level}
                  onChange={(e) =>
                    setForm((prev) => ({ ...prev, level: e.target.value }))
                  }
                  className="w-full rounded-2xl border border-slate-700 bg-slate-950 px-4 py-3 outline-none focus:border-yellow-400"
                >
                  <option>Beginner</option>
                  <option>Intermediate</option>
                  <option>Advanced</option>
                </select>
              </div>

              <div>
                <label className="block text-sm text-slate-400 mb-2">
                  Notes (optional)
                </label>
                <textarea
                  value={form.notes}
                  onChange={(e) =>
                    setForm((prev) => ({ ...prev, notes: e.target.value }))
                  }
                  rows={4}
                  className="w-full rounded-2xl border border-slate-700 bg-slate-950 px-4 py-3 outline-none focus:border-yellow-400"
                />
              </div>

              <button
                type="submit"
                className="rounded-2xl bg-yellow-400 px-6 py-3 font-bold text-slate-950 hover:bg-yellow-300 transition"
              >
                Register selector
              </button>
            </form>
          </div>

          {/* My selectors */}
          <div className="rounded-3xl border border-slate-800 bg-slate-900 p-6 shadow-lg">
            <h2 className="text-2xl font-bold mb-6">My selectors</h2>

            <div className="space-y-4">
              {selectors.map((selector) => (
                <div
                  key={selector.id}
                  className="rounded-2xl border border-slate-800 bg-slate-950 p-5"
                >
                  <h3 className="text-xl font-bold">{selector.name}</h3>
                  <p className="mt-2 text-slate-400">
                    NOVA ID: {selector.novaId} · Age {selector.age}
                  </p>
                  <p className="mt-2 text-slate-300">{selector.experience}</p>

                  <div className="mt-4 flex flex-wrap gap-2">
                    <span className="rounded-full bg-yellow-500/20 px-3 py-1 text-xs font-bold text-yellow-300">
                      {selector.level}
                    </span>

                    {selector.novaActive && (
                      <span className="rounded-full bg-green-500/20 px-3 py-1 text-xs font-bold text-green-300">
                        NOVA Active
                      </span>
                    )}

                    {selector.assignedAssignmentId && (
                      <span className="rounded-full bg-blue-500/20 px-3 py-1 text-xs font-bold text-blue-300">
                        Assigned: #{getAssignmentNumber(selector.assignedAssignmentId)}
                      </span>
                    )}
                  </div>

                  <div className="mt-4 flex flex-wrap gap-3">
                    <button
                      onClick={() => setShowNovaModal(true)}
                      className="rounded-xl border border-slate-700 px-4 py-2 text-sm font-semibold hover:border-yellow-400 transition"
                    >
                      Activate NOVA
                    </button>
                    <button
                      onClick={() => setShowAssignModal(true)}
                      className="rounded-xl border border-slate-700 px-4 py-2 text-sm font-semibold hover:border-yellow-400 transition"
                    >
                      Assign
                    </button>
                    <button
                      onClick={() => setShowLogModal(true)}
                      className="rounded-xl border border-slate-700 px-4 py-2 text-sm font-semibold hover:border-yellow-400 transition"
                    >
                      Log Session
                    </button>
                  </div>
                </div>
              ))}
            </div>
          </div>
        </div>

        {/* Sessions panel */}
        <div className="rounded-3xl border border-slate-800 bg-slate-900 p-6 shadow-lg">
          <h2 className="text-2xl font-bold mb-6">Recent Sessions</h2>

          {sessions.length === 0 ? (
            <div className="rounded-2xl border border-slate-800 bg-slate-950 p-6 text-slate-400">
              No sessions logged yet.
            </div>
          ) : (
            <div className="space-y-4">
              {sessions.map((session) => (
                <SessionCard key={session.id} session={session} />
              ))}
            </div>
          )}
        </div>
      </div>

      <AssignAssignmentModal
        open={showAssignModal}
        onClose={() => setShowAssignModal(false)}
      />
      <ActivateNovaModal
        open={showNovaModal}
        onClose={() => setShowNovaModal(false)}
      />
      <LogSessionModal
        open={showLogModal}
        onClose={() => setShowLogModal(false)}
      />
    </div>
  );
}
