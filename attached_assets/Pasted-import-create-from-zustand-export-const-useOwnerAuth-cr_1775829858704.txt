🔐 2️⃣ OWNER LOCK WRAPPER
src/components/OwnerLock.jsx
import { useState } from "react";
import { useOwnerAuth } from "../store/ownerAuthStore";

export default function OwnerLock({ children }) {
  const { isOwnerAuthenticated, login } = useOwnerAuth();

  const [username, setUsername] = useState("");
  const [password, setPassword] = useState("");

  if (!isOwnerAuthenticated) {
    return (
      <div className="min-h-screen flex items-center justify-center bg-slate-950 text-white">
        <div className="bg-slate-900 p-8 rounded-2xl w-full max-w-sm">
          <h2 className="text-xl font-bold text-yellow-400 mb-4">
            Owner Access
          </h2>

          <input
            placeholder="Username"
            className="w-full p-3 mb-3 bg-slate-800 rounded"
            onChange={(e) => setUsername(e.target.value)}
          />

          <input
            type="password"
            placeholder="Password"
            className="w-full p-3 mb-4 bg-slate-800 rounded"
            onChange={(e) => setPassword(e.target.value)}
          />

          <button
            onClick={() => {
              const success = login(username, password);
              if (!success) alert("Invalid credentials");
            }}
            className="w-full bg-yellow-400 text-black py-3 rounded-xl font-bold"
          >
            Enter
          </button>
        </div>
      </div>
    );
  }

  return children;
}
🔒 3️⃣ AUTO LOCK ON EXIT

Add this to your main layout or App.jsx:

import { useEffect } from "react";
import { useOwnerAuth } from "./store/ownerAuthStore";

function useAutoLock() {
  const { logout } = useOwnerAuth();

  useEffect(() => {
    const handleUnload = () => {
      logout();
    };

    window.addEventListener("beforeunload", handleUnload);

    return () => {
      window.removeEventListener("beforeunload", handleUnload);
    };
  }, []);
}

Then call:

useAutoLock();
