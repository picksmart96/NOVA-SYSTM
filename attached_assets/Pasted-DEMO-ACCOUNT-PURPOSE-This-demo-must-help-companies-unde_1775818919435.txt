4) Update src/lib/accessControl.js
export function canAccessPage(user, pageKey) {
  const publicPages = [
    "home",
    "pricing",
    "choose-plan",
    "checkout-personal",
    "checkout-company",
    "owner-access",
    "privacy",
    "terms",
    "demo",
    "demo-training",
    "demo-leaderboard",
    "demo-nova-help",
    "demo-nova-trainer",
    "demo-trainer-dashboard",
    "demo-supervisor-dashboard",
  ];

  if (publicPages.includes(pageKey)) return true;

  if (!user) return false;

  if (user.role === "owner") return true;

  if (user.isDemoUser) {
    if (user.role === "trainer") {
      return [
        "training",
        "nova-help",
        "nova-trainer",
        "common-mistakes",
        "leaderboard",
        "selector-breaking-news",
        "trainer-dashboard",
      ].includes(pageKey);
    }

    if (user.role === "supervisor") {
      return [
        "training",
        "nova-help",
        "nova-trainer",
        "common-mistakes",
        "leaderboard",
        "selector-breaking-news",
        "trainer-dashboard",
        "supervisor-dashboard",
      ].includes(pageKey);
    }

    return [
      "training",
      "nova-help",
      "nova-trainer",
      "common-mistakes",
      "leaderboard",
      "selector-breaking-news",
    ].includes(pageKey);
  }

  if (!user.isSubscribed) return false;

  if (pageKey === "users-access" || pageKey === "owner") return false;

  if (user.subscriptionPlan === "personal") {
    return [
      "training",
      "nova-help",
      "nova-trainer",
      "common-mistakes",
      "leaderboard",
      "selector-breaking-news",
    ].includes(pageKey);
  }

  if (user.subscriptionPlan === "company") {
    if (user.role === "selector") {
      return [
        "training",
        "nova-help",
        "nova-trainer",
        "common-mistakes",
        "leaderboard",
        "selector-breaking-news",
      ].includes(pageKey);
    }

    if (user.role === "trainer") {
      return [
        "training",
        "nova-help",
        "nova-trainer",
        "common-mistakes",
        "leaderboard",
        "selector-breaking-news",
        "trainer-dashboard",
      ].includes(pageKey);
    }

    if (user.role === "supervisor") {
      return [
        "training",
        "nova-help",
        "nova-trainer",
        "common-mistakes",
        "leaderboard",
        "selector-breaking-news",
        "trainer-dashboard",
        "supervisor-dashboard",
      ].includes(pageKey);
    }
  }

  return false;
}
